module MyModule.Math
{
    pi = 3.14;

    odd = lambda(n) => { 2 * n + 1; };
    even = lambda(n) => { 2 * n; };
    sum = lambda(l) => {
            l = l;
            s = 0;
            a = 0;
            b = len(l);
            while(a < b){
                s += l[a];
                a += 1;
            }
            s;
    };

    E_M = lambda(n) => {
        i = [1..n];
        df = lambda(x, y, z) => {x == y and y == z;};
        fi = lambda(x, y) => {x = x; y = y; [gen df(x, y, j) for j in i end];};
        [gen fi(x, x) for x in i end];
    };

    fibo_sum = lambda (n) => {
            if(n == 1 or n == 2)
                1;
            else
                if(n <= 0)
                    -1;
                else {
                    a11 = 1; b11 = 1; x11 = 2;
                    until(x11 == n){
                        c11 = a11 + b11;
                        a11 = b11;
                        b11 = c11;
                        x11 = x11 + 1;
                    }
                    c11;
                }
    };
    E_4 = E_M(4);
}
