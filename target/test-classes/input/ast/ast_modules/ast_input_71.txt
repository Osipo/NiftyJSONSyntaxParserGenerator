module MyModule.Math
{
    pi = 3.14;
    sum = lambda(l) => {
            l = l;
            s = 0;
            a = 0;
            b = len(l);
            while(a < b){
                s += l[a];
                a += 1;
            }
            s;
    };

    fibo_sum = lambda (n) => {
            if(n == 1 or n == 2)
                1;
            else
                if(n <= 0)
                    -1;
                else {
                    a11 = 1; b11 = 1; x11 = 2;
                    until(x11 == n){
                        c11 = a11 + b11;
                        a11 = b11;
                        b11 = c11;
                        x11 = x11 + 1;
                    }
                    c11;
                }
    };
}
